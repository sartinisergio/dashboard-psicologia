            <div class="collapsible-header" onclick="toggleC(this)">6. Limiti dell'analisi<span class="arrow">▼</span></div>
            <div class="collapsible-body"><div class="cc">
                <p>L'analisi risente di tre limiti principali.         r.style.display=(!s||t.includes(s))&&(!e||t.includes(e))&&classMatch?'':'none';
    });
}

// Table sort
let sortDir={};
function sortTable(c){
    const tb=document.querySelector('#mainTable tbody');
    const rows=Array.from(tb.querySelectorAll('tr'));
    sortDir[c]=!sortDir[c];
    rows.sort((a,b)=>{
        const at=a.children[c].textContent.trim().toLowerCase();
        const bt=b.children[c].textContent.trim().toLowerCase();
        return sortDir[c]?at.localeCompare(bt):bt.localeCompare(at);
    });
    rows.forEach(r=>tb.appendChild(r));
}
